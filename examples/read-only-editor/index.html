() => {
  const [editorState, setEditorState] = React.useState(
    MUIEditorState.createEmpty(),
  )
  const onChange = newState => {
    setEditorState(newState)
  }
  return (
    <MUIEditor
      editorState={editorState}
      onChange={onChange}
      config={{ draftEditor: { readOnly: true } }}
    />
  )
}
